Wheel Spinner App

// Define an array of names (this will be controlled by user input in a real app) const names = ["Alice", "Bob", "Charlie", "David", "Eva", "Frank"]; // Setup for the wheel const wheel = document.getElementById('wheel'); const needle = document.getElementById('needle'); const spinButton = document.getElementById('spinButton'); const resultDiv = document.getElementById('result'); // Function to generate the slices based on names function createWheel() { const sliceAngle = 360 / names.length; // Calculate angle per slice names.forEach((name, index) => { const slice = document.createElement('div'); slice.classList.add('slice'); slice.innerText = name; slice.style.transform = `rotate(${sliceAngle * index}deg)`; wheel.appendChild(slice); }); } // Function to spin the wheel and control the outcome let isSpinning = false; spinButton.addEventListener('click', function() { if (isSpinning) return; // Prevent multiple spins at once isSpinning = true; const randomNameIndex = Math.floor(Math.random() * names.length); // You can control this to make the name predictable const targetAngle = 360 * 5 + (randomNameIndex * (360 / names.length)); // 5 spins to make it more dramatic // Animate wheel spinning wheel.style.transition = 'transform 4s ease-out'; wheel.style.transform = `rotate(${targetAngle}deg)`; // After spinning stops, show the result setTimeout(() => { isSpinning = false; const selectedName = names[randomNameIndex]; resultDiv.innerText = `The result is: ${selectedName}`; }, 4000); // Match the duration of the spin animation }); // Initialize the wheel when the page loads createWheel();
